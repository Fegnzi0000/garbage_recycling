使用微信开发者工具获取真实Code进行后端接口测试

使用说明
1

在微信开发者工具中创建一个小程序项目

2

在小程序页面中添加以下代码：

// 获取微信登录Code
wx.login({
success: (res) => {
if (res.code) {
console.log('微信登录Code:', res.code);
// 复制此Code到下面的输入框中
}
}
})
3

将获取到的Code粘贴到下面的输入框中：

4

填写其他测试参数并发送请求

请求参数

小程序代码示例

在微信开发者工具中，创建一个页面并添加以下代码：

// pages/login/login.js
Page({
onLoad: function() {
this.getLoginCode();
},

getLoginCode: function() {
wx.login({
success: (res) => {
if (res.code) {
console.log('微信登录Code:', res.code);
// 复制这个code到测试工具中
// 注意：这个code5分钟有效，且只能使用一次
} else {
console.log('登录失败！' + res.errMsg);
}
}
});
},

copyCode: function() {
wx.setClipboardData({
data: this.data.code,
success: () => {
wx.showToast({
title: 'Code已复制'
});
}
});
}
})

WXML页面代码：

<!--pages/login/login.wxml-->
<view class="container">
<text>微信登录Code获取示例</text>
<button bindtap="getLoginCode">获取登录Code</button>
<view wx:if="{{code}}">
<text>获取到的Code: {{code}}</text>
<button bindtap="copyCode">复制Code</button>
</view>
</view>
